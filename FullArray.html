方法一

function swap(i,j) {
		var temp;
		temp = myArr1[i];
		myArr1[i]=myArr1[j];
		myArr1[j]=temp;
        
		
		recStr[recStrn] = ("s"+myArr1[i])+myArr1[j]; //记录步骤
		recStrn++;
	}
function fullArray1(m,n) {
	if (m==n) {
		recStr[recStrn] = "p"; //记录步骤
		for (var i=0;i<=n;i++) {
			recStr[recStrn] +=myArr1[i];
		}
		recStrn++;
				
		return;
	}
	else {
		for (var j=m;j<=n;j++) {
			swap(m,j);
			fullArray1(m+1,n);
			swap(j,m);
		}
		return;
	}
}

方法二

function searchNum(n) {
		if (recArr[n]==0) return true;
		else return false;
	}
function fullArray2(n,max) {
	if (n>max) {
		recStr[recStrn] = "p"; //记录步骤
		for (var i=0;i<n-1;i++) {
			recStr[recStrn] +=myArr2[i];
		}
		recStrn++;
		return;
	}
	for (var i=1;i<=max;i++)　{
		if (searchNum(i)) {
			recArr[i]=n; //记录步骤
			recStr[recStrn] = ("a"+i)+n;
			recStrn++;
			myArr2[n-1]=i;
			fullArray2(n+1,max);
			recStr[recStrn] = "r"+i; //记录步骤
			recStrn++;
			recArr[i]=0;
		}
	}
    return;
}
